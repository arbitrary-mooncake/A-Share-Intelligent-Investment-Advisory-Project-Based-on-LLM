# 基金专区 (Fund Zone) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add "基金专区" as the 5th major feature alongside 股票查询/智能问答/股票池/批量打分, with 7 parallel analysis agents + merge node + report/scoring agents integrated into the LangGraph ecosystem.

**Architecture:** Extend the existing two-phase agent pattern (parallel MCP data fetch → LLM analysis) to 7 fund analysis agents. Create a separate fund LangGraph workflow (`fund_main.py`) that shares AgentState, MCP client, caching, and model config with the stock workflow. Add 8 fund-specific MCP tools to `tushare_mcp_server.py`.

**Tech Stack:** LangGraph, FastMCP, Tushare Pro API, Streamlit, FastAPI, existing OpenAI-compatible LLM client

# ── 基金专用数据函数 ──

def ts_fund_basic(code: str) -> list:
    """基金基本信息"""
    ts = _ts_code(code)
    return _dicts(_call("fund_basic", {"ts_code": ts},
                        "ts_code,name,management,custodian,fund_type,found_date,"
                        "due_date,list_date,issue_date,issue_amount,m_fee,c_fee,"
                        "duration_year,p_value,min_amount,exp_return,benchmark,"
                        "status,invest_type,type,trustee,purc_startdate,redm_startdate,market"))

def ts_fund_nav(code: str, days: int = 500) -> list:
    """基金净值历史"""
    s = (_datetime.now() - _timedelta(days=days + 30)).strftime("%Y%m%d")
    e = _datetime.now().strftime("%Y%m%d")
    ts = _ts_code(code)
    return _dicts(_call("fund_nav", {"ts_code": ts, "start_date": s, "end_date": e},
                        "ts_code,nav_date,unit_nav,accum_nav,accum_div,adj_nav"))

def ts_fund_portfolio(code: str) -> list:
    """基金持仓（最新一期）"""
    ts = _ts_code(code)
    return _dicts(_call("fund_portfolio", {"ts_code": ts},
                        "ts_code,ann_date,end_date,symbol,mkv,amount,stk_mkv_ratio,stk_float_ratio"))

def ts_fund_manager(code: str) -> list:
    """基金经理信息"""
    ts = _ts_code(code)
    items = _dicts(_call("fund_manager", {"ts_code": ts},
                         "ts_code,ann_date,name,gender,birth_year,edu,nationality,begin_date,end_date,resume"))
    return items

def ts_fund_div(code: str) -> list:
    """基金分红记录"""
    ts = _ts_code(code)
    return _dicts(_call("fund_div", {"ts_code": ts},
                        "ts_code,ann_date,imp_anndate,base_date,div_proc,record_date,"
                        "ex_date,pay_date,div_cash,base_unit,ear_distr,ear_amount"))

def ts_fund_share(code: str) -> list:
    """基金份额变动"""
    ts = _ts_code(code)
    s = (_datetime.now() - _timedelta(days=365)).strftime("%Y%m%d")
    e = _datetime.now().strftime("%Y%m%d")
    return _dicts(_call("fund_share", {"ts_code": ts, "start_date": s, "end_date": e},
                        "ts_code,trade_date,fd_share"))

def ts_fund_company(code: str) -> list:
    """基金管理人信息（通过fund_basic获取management后查询）"""
    # 先获取基金基本信息中的管理人
    basic = ts_fund_basic(code)
    if not basic:
        return []
    mgmt = basic[0].get("management", "")
    # 搜索管理公司
    companies = _dicts(_call("fund_company", {},
                             "name,shortname,province,city,setup_date,employees,reg_capital,chairman,manager,website"))
    if mgmt and companies:
        return [c for c in companies if mgmt in c.get("name", "")]
    return []
```

@app.tool()
def tushare_fund_basic(code: str) -> str:
    """获取公募基金/ETF基本信息: 基金名称、类型、管理人、托管人、业绩基准、费率、成立日期、申赎日期、投资类型"""

    return _format_result(items) if items else f"未找到基金 {code} 的信息"


@app.tool()
def tushare_fund_nav(code: str, days: int = 500) -> str:
    """获取公募基金净值历史: 单位净值、累计净值、累计分红、复权净值。days默认500个交易日"""
    items = ts_fund_nav(code, days=days)
    return _format_result(items, max_rows=200) if items else f"未找到基金 {code} 的净值数据"


@app.tool()
def tushare_fund_portfolio(code: str) -> str:
    """获取公募基金持仓明细: 最新一期持仓股票代码、市值、占净值比例、占流通股比例"""
    items = ts_fund_portfolio(code)
    return _format_result(items, max_rows=100) if items else f"未找到基金 {code} 的持仓数据"


@app.tool()
def tushare_fund_manager(code: str) -> str:
    """获取基金经理信息: 姓名、性别、学历、任职起始日、任职结束日、个人简历"""
    items = ts_fund_manager(code)
    return _format_result(items, max_rows=10) if items else f"未找到基金 {code} 的经理信息"


@app.tool()
def tushare_fund_div(code: str) -> str:
    """获取基金分红记录: 公告日、除权日、分红金额、基准份额"""
    items = ts_fund_div(code)
    return _format_result(items, max_rows=30) if items else f"未找到基金 {code} 的分红记录"


@app.tool()
def tushare_fund_share(code: str) -> str:
    """获取基金份额变动: 近一年每日份额数据，用于判断规模变化趋势"""
    items = ts_fund_share(code)
    return _format_result(items, max_rows=200) if items else f"未找到基金 {code} 的份额数据"


@app.tool()
def tushare_fund_company(code: str) -> str:
    """获取基金管理人（基金公司）信息: 公司名称、成立日期、员工数、注册资本、董事长、总经理"""
    items = ts_fund_company(code)
    return _format_result(items) if items else f"未找到基金 {code} 的管理人信息"


@app.tool()
def tushare_fund_search(keyword: str) -> str:
    """按名称搜索公募基金: 输入关键词(如'沪深300''科创50')，返回匹配的基金代码、名称、类型"""
    # 搜索场内基金 (market=E)
    items_e = _dicts(_call("fund_basic", {"market": "E"},
                           "ts_code,name,management,fund_type,found_date,benchmark,invest_type,type"))
    # 搜索场外基金 (market=O)，limit取100避免超量
    items_o = _dicts(_call("fund_basic", {"market": "O"},
                           "ts_code,name,management,fund_type,found_date,benchmark,invest_type,type"))
    all_items = (items_e or []) + (items_o or [])
    if not all_items:
        return "未找到任何基金数据"
    kw = keyword.strip().lower()
    matched = [d for d in all_items if kw in d.get("name", "").lower() or kw in d.get("ts_code", "").lower()]
    if not matched:
        return f"未找到名称包含'{keyword}'的基金"
    return _format_result(matched[:30], max_rows=30)
```

### Phase 4: LangGraph Workflow & CLI

### Task 12: Create fund_main.py

**Files:**
- Create: `src/fund_main.py`

Register all 7 fund agents + merge_node + report/scoring as LangGraph nodes. Build the workflow:
```python
workflow = StateGraph(AgentState)
workflow.add_node("start_node", lambda state: state)
# 7 fund analysis agents
for name, agent in fund_agents.items():
    workflow.add_node(name, agent)
workflow.add_node("fund_merge_node", fund_merge_node)
workflow.add_node("fund_report_agent", fund_report_agent)
workflow.add_node("fund_scoring_agent", fund_scoring_agent)

# edges: start → 7 agents → merge → [report / scoring] → END
workflow.set_entry_point("start_node")
for name in fund_agents:
    workflow.add_edge("start_node", name)
    workflow.add_edge(name, "fund_merge_node")
# merge → report/scoring based on user intent
workflow.add_conditional_edges("fund_merge_node", route_intent, {
    "report": "fund_report_agent",
    "score": "fund_scoring_agent",
    "both": "__both__"
})
workflow.add_edge("fund_report_agent", END)
workflow.add_edge("fund_scoring_agent", END)
```

Support CLI: `python -m src.fund_main "分析基金000001"` and `python -m src.fund_main score 000001`

Support fund code normalization: 6-digit codes without prefix handled (51/58/15/16/18 detected as fund/ETF)
